def Image2array(imgPath):
    Image = imgPath
    Img1 = cv2.imread(Image)[...,::-1]
    Img = cv2.resize(Img1,(512,512))
    return Img

labels = ['hijab_images', 'medical_cap_images','turban_images','winter_hat_images']
img_size = 512
def get_data(data_dir):
    data = [] 
    for label in labels: 
        path = os.path.join(data_dir, label)
        class_num = labels.index(label)
        print(class_num)
        for img in os.listdir(path):
            try:
                img_arr = Image2array(os.path.join(path, img)) #convert BGR to RGB format
                #resized_arr = cv2.resize(img_arr, (img_size, img_size)) # Reshaping images to preferred size
                data.append([img_arr, class_num])
            except Exception as e:
                print(e)
    return np.array(data)
